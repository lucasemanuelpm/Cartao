document.addEventListener('DOMContentLoaded', () => { const nome = localStorage.getItem('nome') || 'Nome não informado'; const email = localStorage.getItem('email') || 'Email não informado'; const telefone = localStorage.getItem('telefone') || 'Telefone não informado'; const habilidades = JSON.parse(localStorage.getItem('habilidades')) || []; const requisitos = JSON.parse(localStorage.getItem('requisitos')) || []; // Calcula compatibilidade (lógica simples para exemplo) const compatibilidade = requisitos.every(req => habilidades.includes(req)); // Cria o cartão com as informações e o resultado da compatibilidade criaCartao(nome, email, telefone, habilidades, requisitos, compatibilidade); }); function criaCartao(nome, email, telefone, habilidades, requisitos, compatibilidade) { let container = document.getElementById('container'); let cartao = document.createElement('article'); cartao.className = 'cartao'; cartao.innerHTML = `

${nome}

Email: ${email}

Telefone: ${telefone}

Habilidades: ${habilidades.join(', ')}

Requisitos: ${requisitos.join(', ')}

Compatibilidade: ${compatibilidade ? 'Sim' : 'Não'}

`; container.appendChild(cartao); }
